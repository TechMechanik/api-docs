Example 1

Request:

<soapenv:Envelope xmlns:soapenv="http://schemas.xmlsoap.org/soap/envelope/" xmlns:web="http://example.com/webservices">
    <soapenv:Header/>
    <soapenv:Body>
        <web:MyRequest>
            <web:Parameter>Value</web:Parameter>
        </web:MyRequest>
    </soapenv:Body>
</soapenv:Envelope>

    <soapenv:Body>

        </web:MyResponse>
    </soapenv:Body>
</soapenv:Envelope>
